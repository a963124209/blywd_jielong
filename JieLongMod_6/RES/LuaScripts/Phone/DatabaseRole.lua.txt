            end
        })

        for i, msg in ipairs(role.daily) do
            triggerMgr.addTrigger({
                id = "角色_每日闲聊_" .. i .. "_" .. role.roleId,
                type = "daily_chat",
                weight = 10,
                once = false,
                priority = "normal",
                bindRole = role.roleId,

                end
            })
        end

            id = "角色_首领闲聊_" .. role.roleId,
            type = "daily_chat",

            once = false,
            priority = "normal",
            bindRole = role.roleId,
            require = database.InPlayerCamp,
            event = function(context)
